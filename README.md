                { from: 'git', to: 'mysql' },
                { from: 'php', to: 'mysql' },
                { from: 'tailwind', to: 'wordpress' },
                { from: 'php', to: 'wordpress' },
                { from: 'mysql', to: 'wordpress' }
            ];

            const roadmap = document.getElementById('roadmap');
            let scale = 1;
            let pos = { x: 0, y: 0 };
            let isDragging = false;
            let startX, startY;
            let initialPinchDistance = null;
            let currentNode = null;

            // নোড তৈরি করুন
            roadmapData.forEach(node => {
                const nodeEl = document.createElement('div');
                nodeEl.className = `node node-${node.level}`;
                nodeEl.id = `node-${node.id}`;
                nodeEl.style.left = `${node.x}px`;
                nodeEl.style.top = `${node.y}px`;
                
                const levelLabels = {
                    'beginner': 'শুরু',
                    'intermediate': 'মধ্যবর্তী',
                    'advanced': 'উন্নত',
                    'expert': 'বিশেষজ্ঞ'
                };
                
                nodeEl.innerHTML = `
                    <span class="level-indicator">${levelLabels[node.level]}</span>
                    <input type="checkbox" class="progress-checkbox">
                    <div class="font-semibold">${node.title}</div>
                    <div class="text-xs mt-2"><i class="fas fa-click"></i> ক্লিক করুন বিস্তারিত দেখতে</div>
                `;
                roadmap.appendChild(nodeEl);

                // নোড ক্লিক ইভেন্ট
                nodeEl.addEventListener('click', (e) => {
                    if (e.target.type !== 'checkbox') {
                        showNodeContent(node);
                    }
                });

                // প্রগ্রেস চেকবক্স ইভেন্ট
                const checkbox = nodeEl.querySelector('.progress-checkbox');
                checkbox.addEventListener('change', function() {
                    nodeEl.classList.toggle('completed', this.checked);
                    saveProgress(node.id, this.checked);
                });

                // স্থানীয় স্টোরেজ থেকে প্রগতি লোড করুন
                const isCompleted = localStorage.getItem(`progress-${node.id}`) === 'true';
                checkbox.checked = isCompleted;
                if (isCompleted) {
                    nodeEl.classList.add('completed');
                }
            });

            // সংযোগ তৈরি করুন
            connections.forEach(conn => {
                const fromNode = roadmapData.find(n => n.id === conn.from);
                const toNode = roadmapData.find(n => n.id === conn.to);
                
                if (fromNode && toNode) {
                    const line = document.createElement('div');
                    line.className = 'connection';
                    
                    const dx = toNode.x - fromNode.x;
                    const dy = toNode.y - fromNode.y;
                    const length = Math.sqrt(dx * dx + dy * dy);
                    const angle = Math.atan2(dy, dx) * 180 / Math.PI;
                    
                    line.style.width = `${length}px`;
                    line.style.left = `${fromNode.x + 90}px`;
                    line.style.top = `${fromNode.y + 25}px`;
                    line.style.transform = `rotate(${angle}deg)`;
                    
                    roadmap.appendChild(line);
                    
                    // সংযোগের জন্য তীর যোগ করুন
                    const arrow = document.createElement('div');
                    arrow.className = 'connection-arrow';
                    arrow.style.left = `${toNode.x - 10}px`;
                    arrow.style.top = `${toNode.y + 25 - 5}px`;
                    arrow.style.transform = `rotate(${angle}deg)`;
                    roadmap.appendChild(arrow);
                }
            });

            // নোড বিষয়বস্তু প্রদর্শন
            function showNodeContent(node) {
                // বিদ্যমান কন্টেন্ট বন্ধ করুন
                const existingContent = document.querySelector('.node-content');
                if (existingContent) {
                    existingContent.remove();
                }
                
                // নতুন কন্টেন্ট তৈরি করুন
                const content = document.createElement('div');
                content.className = 'node-content';
                content.innerHTML = `
                    <span class="close-btn">&times;</span>
                    ${node.content}
                `;
                
                document.body.appendChild(content);
                
                // কন্টেন্ট অবস্থান
                const rect = roadmap.getBoundingClientRect();
                content.style.left = `${window.innerWidth/2 - 160}px`;
                content.style.top = `${window.innerHeight/2 - 200}px`;
                
                // বন্ধ বাটন ইভেন্ট
                content.querySelector('.close-btn').addEventListener('click', function() {
                    content.remove();
                });
                
                // বাইরে ক্লিক করলে বন্ধ করুন
                content.addEventListener('click', function(e) {
                    if (e.target === content) {
                        content.remove();
                    }
                });
            }

            // প্রগতি সংরক্ষণ করুন
            function saveProgress(nodeId, completed) {
                localStorage.setItem(`progress-${nodeId}`, completed);
            }

            // ড্র্যাগ ফাংশন
            roadmap.addEventListener('mousedown', function(e) {
                if (e.button === 0) { // শুধুমাত্র বাম মাউস বাটন
                    isDragging = true;
                    startX = e.clientX - pos.x;
                    startY = e.clientY - pos.y;
                    roadmap.style.cursor = 'grabbing';
                }
            });

            document.addEventListener('mousemove', function(e) {
                if (isDragging) {
                    pos.x = e.clientX - startX;
                    pos.y = e.clientY - startY;
                    updateTransform();
                }
            });

            document.addEventListener('mouseup', function() {
                isDragging = false;
                roadmap.style.cursor = 'grab';
            });

            // টাচ ইভেন্টগুলি (মোবাইলের জন্য)
            roadmap.addEventListener('touchstart', function(e) {
                if (e.touches.length === 1) {
                    isDragging = true;
                    startX = e.touches[0].clientX - pos.x;
                    startY = e.touches[0].clientY - pos.y;
                } else if (e.touches.length === 2) {
                    // পিঞ্চ জুম
                    initialPinchDistance = getDistance(
                        e.touches[0].clientX, e.touches[0].clientY,
                        e.touches[1].clientX, e.touches[1].clientY
                    );
                }
            });

            document.addEventListener('touchmove', function(e) {
                if (e.touches.length === 1 && isDragging) {
                    pos.x = e.touches[0].clientX - startX;
                    pos.y = e.touches[0].clientY - startY;
                    updateTransform();
                } else if (e.touches.length === 2 && initialPinchDistance !== null) {
                    // পিঞ্চ জুম
                    const currentDistance = getDistance(
                        e.touches[0].clientX, e.touches[0].clientY,
                        e.touches[1].clientX, e.touches[1].clientY
                    );
                    
                    const delta = currentDistance - initialPinchDistance;
                    scale += delta * 0.01;
                    scale = Math.min(Math.max(0.5, scale), 3);
                    initialPinchDistance = currentDistance;
                    updateTransform();
                }
            });

            document.addEventListener('touchend', function(e) {
                if (e.touches.length === 0) {
                    isDragging = false;
                    initialPinchDistance = null;
                }
            });

            // দুটি টাচ পয়েন্টের মধ্যে দূরত্ব পান
            function getDistance(x1, y1, x2, y2) {
                return Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
            }

            // জুম ইন/আউট বাটন
            document.getElementById('zoom-in').addEventListener('click', function() {
                scale += 0.2;
                scale = Math.min(scale, 3);
                updateTransform();
            });

            document.getElementById('zoom-out').addEventListener('click', function() {
                scale -= 0.2;
                scale = Math.max(scale, 0.5);
                updateTransform();
            });

            // ভিউ রিসেট করুন
            document.getElementById('reset-view').addEventListener('click', function() {
                scale = 1;
                pos.x = 0;
                pos.y = 0;
                updateTransform();
            });

            // রূপান্তর আপডেট করুন
            function updateTransform() {
                roadmap.style.transform = `translate(${pos.x}px, ${pos.y}px) scale(${scale})`;
            }

            // প্রারম্ভিক অবস্থান সেট করুন
            pos.x = roadmap.offsetWidth / 2 - 450;
            pos.y = roadmap.offsetHeight / 2 - 200;
            updateTransform();
        });
    </script>
